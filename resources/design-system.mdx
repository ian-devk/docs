## Design Philosophy

### Core Principles

Our design system is built on four foundational principles that guide every design decision:

<Accordion title="Safety-First Design">
**Emergency Optimization**: Every component is designed to function under stress, with large touch targets (minimum 44px), high contrast ratios (4.5:1+), and clear visual hierarchy that guides users to critical actions.

**Stress-Tested Usability**: Interface elements are tested in simulated emergency scenarios to ensure functionality when users are under pressure, distracted, or in low-light conditions.

**Fault-Tolerant Interactions**: Multiple confirmation paths and clear undo mechanisms prevent accidental actions while maintaining speed for genuine emergencies.
</Accordion>

<Accordion title="Trustworthy Aesthetics">
**Professional Confidence**: Clean, modern design language that builds user trust through consistent application of design tokens and predictable interaction patterns.

**Emotional Intelligence**: Calming color gradients and warm interface elements reduce anxiety while maintaining alertness for safety features.

**Brand Reliability**: Consistent visual identity that reinforces Shelther's commitment to user safety across all touchpoints.
</Accordion>

<Accordion title="Universal Accessibility">
**Inclusive by Design**: Complete WCAG 2.1 AA compliance with support for screen readers, motor accessibility, and cognitive accessibility considerations.

**Multi-Modal Interaction**: Support for voice commands, gesture controls, and traditional touch interfaces to accommodate diverse user needs.

**Cultural Sensitivity**: Color choices and iconography that work across cultural contexts without relying on cultural assumptions.
</Accordion>

<Accordion title="Performance Excellence">
**Instant Response**: Optimized for &lt;200ms standard interactions and &lt;500ms emergency response times with predictable performance across device ranges.

**Battery Efficiency**: Power-optimized animations and background processes that don't compromise device longevity during extended safety monitoring.

**Network Resilience**: Graceful degradation for low-connectivity scenarios with offline-first critical functions.
</Accordion>

```typescript React Native Implementation
// components/Button/Button.tsx
import React from 'react';
import { 
  TouchableOpacity, 
  Text, 
  View, 
  StyleSheet, 
  ViewStyle, 
  TextStyle 
} from 'react-native';
import { tokens } from '../tokens';

export interface ButtonProps {
  variant?: 'primary' | 'secondary' | 'emergency' | 'ghost';
  size?: 'sm' | 'md' | 'lg' | 'xl';
  disabled?: boolean;
  loading?: boolean;
  icon?: React.ReactNode;
  iconPosition?: 'left' | 'right';
  onPress?: () => void;
  children: React.ReactNode;
  testID?: string;
  accessibilityLabel?: string;
  accessibilityHint?: string;
}

const Button: React.FC<ButtonProps> = ({
  variant = 'primary',
  size = 'md',
  disabled = false,
  loading = false,
  icon,
  iconPosition = 'left',
  onPress,
  children,
  testID,
  accessibilityLabel,
  accessibilityHint,
}) => {
  const buttonStyle: ViewStyle = [
    styles.base,
    styles[variant],
    styles[size],
    disabled && styles.disabled,
    loading && styles.loading,
  ].filter(Boolean);

  const textStyle: TextStyle = [
    styles.text,
    styles[`${variant}Text`],
    styles[`${size}Text`],
    disabled && styles.disabledText,
  ].filter(Boolean);

  return (
    <TouchableOpacity
      style={buttonStyle}
      onPress={onPress}
      disabled={disabled || loading}
      activeOpacity={0.8}
      testID={testID}
      accessibilityRole="button"
      accessibilityLabel={accessibilityLabel}
      accessibilityHint={accessibilityHint}
      accessibilityState={{ disabled }}
    >
      <View style={styles.content}>
        {icon && iconPosition === 'left' && (
          <View style={styles.iconLeft}>{icon}</View>
        )}
        
        <Text style={textStyle}>{children}</Text>
        
        {icon && iconPosition === 'right' && (
          <View style={styles.iconRight}>{icon}</View>
        )}
        
        {loading && (
          <View style={styles.loadingIndicator}>
            {/* Loading spinner component */}
          </View>
        )}
      </View>
    </TouchableOpacity>
  );
};

const styles = StyleSheet.create({
  base: {
    borderRadius: tokens.borderRadius.default,
    justifyContent: 'center',
    alignItems: 'center',
    flexDirection: 'row',
    minHeight: 44, // WCAG touch target minimum
  },
  
  // Variant styles
  primary: {
    backgroundColor: tokens.color.component.button.primary.background,
    borderWidth: 0,
  },
  secondary: {
    backgroundColor: tokens.color.component.button.secondary.background,
    borderWidth: 1,
    borderColor: tokens.color.component.button.secondary.border,
  },
  emergency: {
    backgroundColor: tokens.color.component.button.emergency.background,
    borderWidth: 0,
    minHeight: 56, // Larger touch target for emergencies
  },
  ghost: {
    backgroundColor: 'transparent',
    borderWidth: 0,
  },
  
  // Size styles
  sm: {
    paddingHorizontal: tokens.spacing.primitive[4],
    paddingVertical: tokens.spacing.primitive[2],
    minHeight: 36,
  },
  md: {
    paddingHorizontal: tokens.spacing.primitive[6],
    paddingVertical: tokens.spacing.primitive[3],
  },
  lg: {
    paddingHorizontal: tokens.spacing.primitive[8],
    paddingVertical: tokens.spacing.primitive[4],
    minHeight: 52,
  },
  xl: {
    paddingHorizontal: tokens.spacing.primitive[10],
    paddingVertical: tokens.spacing.primitive[5],
    minHeight: 56,
  },
  
  // State styles
  disabled: {
    opacity: 0.5,
  },
  loading: {
    opacity: 0.8,
  },
  
  // Content and text styles
  content: {
    flexDirection: 'row',
    alignItems: 'center',
    justifyContent: 'center',
  },
  text: {
    fontFamily: tokens.font.family.primary,
    fontWeight: tokens.font.weight.medium,
    textAlign: 'center',
  },
  primaryText: {
    color: tokens.color.component.button.primary.text,
    fontSize: tokens.font.size.base,
  },
  secondaryText: {
    color: tokens.color.component.button.secondary.text,
    fontSize: tokens.font.size.base,
  },
  emergencyText: {
    color: tokens.color.component.button.emergency.text,
    fontSize: tokens.font.size.lg,
    fontWeight: tokens.font.weight.bold,
  },
  ghostText: {
    color: tokens.color.semantic.text.primary,
    fontSize: tokens.font.size.base,
  },
  disabledText: {
    opacity: 0.6,
  },
  
  // Icon styles
  iconLeft: {
    marginRight: tokens.spacing.primitive[2],
  },
  iconRight: {
    marginLeft: tokens.spacing.primitive[2],
  },
  loadingIndicator: {
    marginLeft: tokens.spacing.primitive[2],
  },
});

export default Button;
```

```typescript Card Component
// components/Card/Card.tsx
import React from 'react';
import { 
  View, 
  TouchableOpacity, 
  StyleSheet, 
  ViewStyle 
} from 'react-native';
import { tokens } from '../tokens';

export interface CardProps {
  variant?: 'default' | 'interactive' | 'elevated';
  padding?: 'sm' | 'md' | 'lg';
  onPress?: () => void;
  disabled?: boolean;
  children: React.ReactNode;
  testID?: string;
  accessibilityLabel?: string;
  accessibilityHint?: string;
  style?: ViewStyle;
}

const Card: React.FC<CardProps> = ({
  variant = 'default',
  padding = 'md',
  onPress,
  disabled = false,
  children,
  testID,
  accessibilityLabel,
  accessibilityHint,
  style,
}) => {
  const cardStyle: ViewStyle = [
    styles.base,
    styles[variant],
    styles[padding],
    disabled && styles.disabled,
    style,
  ].filter(Boolean);

  if (onPress) {
    return (
      <TouchableOpacity
        style={cardStyle}
        onPress={onPress}
        disabled={disabled}
        activeOpacity={0.95}
        testID={testID}
        accessibilityLabel={accessibilityLabel}
        accessibilityHint={accessibilityHint}
        accessibilityRole="button"
      >
        {children}
      </TouchableOpacity>
    );
  }

  return (
    <View style={cardStyle} testID={testID}>
      {children}
    </View>
  );
};

const styles = StyleSheet.create({
  base: {
    backgroundColor: tokens.color.component.card.background,
    borderRadius: tokens.borderRadius.lg,
    borderWidth: 1,
    borderColor: tokens.color.component.card.border,
```

```typescript Theme Provider
// contexts/ThemeContext.tsx
import React, { createContext, useContext, useState, useEffect } from 'react';
import { Appearance, ColorSchemeName } from 'react-native';

interface ThemeContextType {
  theme: 'light' | 'dark' | 'auto';
  colorScheme: ColorSchemeName;
  setTheme: (theme: 'light' | 'dark' | 'auto') => void;
  tokens: typeof lightTokens;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ 
  children 
}) => {
  const [theme, setTheme] = useState<'light' | 'dark' | 'auto'>('auto');
  const [colorScheme, setColorScheme] = useState<ColorSchemeName>(
    Appearance.getColorScheme()
  );

  useEffect(() => {
    const subscription = Appearance.addChangeListener(({ colorScheme }) => {
      setColorScheme(colorScheme);
    });

    return () => subscription?.remove();
  }, []);

  const effectiveScheme = theme === 'auto' ? colorScheme : theme;
  const tokens = effectiveScheme === 'dark' ? darkTokens : lightTokens;

  return (
    <ThemeContext.Provider value={{
      theme,
      colorScheme: effectiveScheme,
      setTheme,
      tokens,
    }}>
      {children}
    </ThemeContext.Provider>
  );
};

export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
};
```

```typescript Performance Monitoring
// utils/performance.ts
import { InteractionManager } from 'react-native';

export class PerformanceMonitor {
  private static instance: PerformanceMonitor;
  private metricsLog: Array<{
    timestamp: number;
    event: string;
    duration?: number;
    metadata?: Record<string, any>;
  }> = [];

  static getInstance(): PerformanceMonitor {
    if (!PerformanceMonitor.instance) {
      PerformanceMonitor.instance = new PerformanceMonitor();
    }
    return PerformanceMonitor.instance;
  }

  logEvent(event: string, metadata?: Record<string, any>) {
    this.metricsLog.push({
      timestamp: Date.now(),
      event,
      metadata,
    });
  }

  measureAsyncOperation<T>(
    operation: () => Promise<T>,
    operationName: string
  ): Promise<T> {
    return new Promise((resolve, reject) => {
      const startTime = Date.now();
      
      InteractionManager.runAfterInteractions(() => {
        operation()
          .then((result) => {
            const duration = Date.now() - startTime;
            this.metricsLog.push({
              timestamp: startTime,
              event: operationName,
              duration,
              metadata: { success: true },
            });
            resolve(result);
          })
          .catch((error) => {
            const duration = Date.now() - startTime;
            this.metricsLog.push({
              timestamp: startTime,
              event: operationName,
              duration,
              metadata: { success: false, error: error.message },
            });
            reject(error);
          });
      });
    });
  }

  getMetrics() {
    return this.metricsLog;
  }

  clearMetrics() {
    this.metricsLog = [];
  }
}
```
