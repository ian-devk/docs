```typescript Component Template
// components/BaseComponent.tsx
import React, { memo, forwardRef } from 'react';
import { View, ViewStyle, AccessibilityProps } from 'react-native';
import { useTheme } from '../contexts/ThemeContext';
import { logger } from '../utils/logger';

interface BaseComponentProps extends AccessibilityProps {
  children?: React.ReactNode;
  style?: ViewStyle;
  testID?: string;
  emergencyMode?: boolean;
  onEmergencyAccess?: () => void;
}

export const BaseComponent = memo(forwardRef<View, BaseComponentProps>(({
  children,
  style,
  testID,
  emergencyMode = false,
  onEmergencyAccess,
  accessible = true,
  accessibilityRole = 'button',
  ...accessibilityProps
}, ref) => {
  const { theme, isHighContrast } = useTheme();
  
  const handlePress = React.useCallback(() => {
    if (emergencyMode && onEmergencyAccess) {
      logger.emergency('Emergency component accessed', { component: testID });
      onEmergencyAccess();
    }
  }, [emergencyMode, onEmergencyAccess, testID]);

  const componentStyle = React.useMemo(() => [
    {
      // Base styles
      padding: theme.spacing.md,
      backgroundColor: emergencyMode 
        ? theme.colors.emergency 
        : theme.colors.surface,
      borderRadius: theme.borderRadius.md,
      // High contrast mode support
      ...(isHighContrast && {
        borderWidth: 2,
        borderColor: theme.colors.primary,
      }),
    },
    style,
  ], [theme, emergencyMode, isHighContrast, style]);

  return (
    <View
      ref={ref}
      style={componentStyle}
      testID={testID}
      accessible={accessible}
      accessibilityRole={accessibilityRole}
      onTouchEnd={emergencyMode ? handlePress : undefined}
      {...accessibilityProps}
    >
      {children}
    </View>
  );
}));

BaseComponent.displayName = 'BaseComponent';
```

```typescript Theme Integration
    success: '#4CAF50',

  },
  typography: {

    h2: { fontSize: 24, fontWeight: 'bold' },
    body: { fontSize: 16, fontWeight: 'normal' },
    caption: { fontSize: 12, fontWeight: 'normal' },
    emergency: { fontSize: 18, fontWeight: 'bold' },
  },
  borderRadius: {
    sm: 4,
    md: 8,
    lg: 16,
  },
  shadows: {
    sm: {
      shadowOffset: { width: 0, height: 1 },
      shadowOpacity: 0.22,
      shadowRadius: 2.22,
      elevation: 3,
    },
    md: {
      shadowOffset: { width: 0, height: 2 },
      shadowOpacity: 0.25,
      shadowRadius: 3.84,
      elevation: 5,
    },
    lg: {
      shadowOffset: { width: 0, height: 4 },
      shadowOpacity: 0.3,
      shadowRadius: 4.65,
      elevation: 8,
    },
  },
};

interface ThemeContextType {
  theme: Theme;
  isDarkMode: boolean;
  isHighContrast: boolean;
  toggleDarkMode: () => void;
  toggleHighContrast: () => void;
}

const ThemeContext = createContext<ThemeContextType | undefined>(undefined);

export const ThemeProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const systemColorScheme = useColorScheme();
  const [isDarkMode, setIsDarkMode] = useState(systemColorScheme === 'dark');
  const [isHighContrast, setIsHighContrast] = useState(false);

  React.useEffect(() => {
    // Check for system-level high contrast setting
    AccessibilityInfo.isHighTextContrastEnabled().then(setIsHighContrast);
  }, []);

  const toggleDarkMode = () => setIsDarkMode(!isDarkMode);
  const toggleHighContrast = () => setIsHighContrast(!isHighContrast);

  const theme = React.useMemo(() => {
    const baseTheme = isDarkMode ? darkTheme : lightTheme;
    
    if (isHighContrast) {
      return {
        ...baseTheme,
        colors: {
          ...baseTheme.colors,
          primary: '#FF0000',
          text: {
            primary: isDarkMode ? '#FFFFFF' : '#000000',
            secondary: isDarkMode ? '#CCCCCC' : '#333333',
            inverse: isDarkMode ? '#000000' : '#FFFFFF',
          },
        },
      };
    }
    
    return baseTheme;
  }, [isDarkMode, isHighContrast]);

  return (
    <ThemeContext.Provider value={{
      theme,
      isDarkMode,
      isHighContrast,
      toggleDarkMode,
      toggleHighContrast,
    }}>
      {children}
    </ThemeContext.Provider>
  );
};

export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within a ThemeProvider');
  }
  return context;
};
```

    ```typescript
    // components/EmergencyButton.tsx
    import React, { useState, useRef } from 'react';
    import { 
      TouchableOpacity, 
      Text, 
      Animated, 
      Vibration,
      AccessibilityInfo 
    } from 'react-native';
    import { useTheme } from '../contexts/ThemeContext';
    import { EmergencyService } from '../services/emergency';
    import { logger } from '../utils/logger';

    interface EmergencyButtonProps {
      onPress?: () => void;
      disabled?: boolean;
      countdown?: boolean;
      countdownDuration?: number;
      testID?: string;
    }

    export const EmergencyButton: React.FC<EmergencyButtonProps> = ({
      onPress,
      disabled = false,
      countdown = true,
      countdownDuration = 5,
      testID = 'emergency-button',
    }) => {
      const { theme } = useTheme();
      const [isPressed, setIsPressed] = useState(false);
      const [countdownActive, setCountdownActive] = useState(false);
      const [timeLeft, setTimeLeft] = useState(countdownDuration);
      
      const scaleAnim = useRef(new Animated.Value(1)).current;
      const pulseAnim = useRef(new Animated.Value(1)).current;
      
      const startCountdown = () => {
        if (disabled) return;
        
        setCountdownActive(true);
        setTimeLeft(countdownDuration);
        
        // Haptic feedback
        Vibration.vibrate([100, 100, 100]);
        
        // Screen reader announcement
        AccessibilityInfo.announceForAccessibility(
          `Emergency alert starting in ${countdownDuration} seconds. Tap again to cancel.`
        );
        
        logger.emergency('Emergency countdown started', { duration: countdownDuration });
        
        const timer = setInterval(() => {
          setTimeLeft((prev) => {
            if (prev <= 1) {
              clearInterval(timer);
              setCountdownActive(false);
              triggerEmergency();
              return 0;
            }
            
            // Vibrate each second
            Vibration.vibrate(200);
            
            // Screen reader countdown
            AccessibilityInfo.announceForAccessibility(`${prev - 1} seconds`);
            
            return prev - 1;
          });
        }, 1000);
        
        // Store timer reference for cleanup
        return timer;
      };
      
      const cancelCountdown = () => {
        setCountdownActive(false);
        setTimeLeft(countdownDuration);
        
        AccessibilityInfo.announceForAccessibility('Emergency alert cancelled');
        logger.info('Emergency countdown cancelled');
      };
      
      const triggerEmergency = async () => {
        try {
          AccessibilityInfo.announceForAccessibility('Emergency alert activated');
          logger.emergency('Emergency button triggered');
          
          await EmergencyService.triggerAlert({
            type: 'manual',
            source: 'emergency_button',
          });
          
          if (onPress) {
            onPress();
          }
        } catch (error) {
          logger.emergency('Emergency button trigger failed', error);
          AccessibilityInfo.announceForAccessibility('Emergency alert failed. Please try again.');
        }
      };
      
      const handlePress = () => {
        if (countdownActive) {
          cancelCountdown();
          return;
        }
        
        if (countdown) {
          startCountdown();
        } else {
          triggerEmergency();
        }
      };
      
      // Button press animation
      const handlePressIn = () => {
        setIsPressed(true);
        Animated.spring(scaleAnim, {
          toValue: 0.95,
          useNativeDriver: true,
        }).start();
      };
      
      const handlePressOut = () => {
        setIsPressed(false);
        Animated.spring(scaleAnim, {
          toValue: 1,
          useNativeDriver: true,
        }).start();
      };
      
      // Pulsing animation during countdown
      React.useEffect(() => {
        if (countdownActive) {
          const pulse = Animated.loop(
            Animated.sequence([
              Animated.timing(pulseAnim, {
                toValue: 1.1,
                duration: 500,
                useNativeDriver: true,
              }),
              Animated.timing(pulseAnim, {
                toValue: 1,
                duration: 500,
                useNativeDriver: true,
              }),
            ])
          );
          pulse.start();
          
          return () => pulse.stop();
        }
      }, [countdownActive, pulseAnim]);
      
      const buttonStyle = [
        {
          width: 120,
          height: 120,
          borderRadius: 60,
          backgroundColor: countdownActive ? theme.colors.warning : theme.colors.emergency,
          justifyContent: 'center',
          alignItems: 'center',
          ...theme.shadows.lg,
        },
        disabled && {
          backgroundColor: theme.colors.text.secondary,
          opacity: 0.5,
        },
      ];
      
      return (
        <Animated.View style={{ transform: [{ scale: scaleAnim }, { scale: pulseAnim }] }}>
          <TouchableOpacity
            style={buttonStyle}
            onPress={handlePress}
            onPressIn={handlePressIn}
            onPressOut={handlePressOut}
            disabled={disabled}
            accessible={true}
            accessibilityRole="button"
            accessibilityLabel={countdownActive ? "Cancel emergency alert" : "Emergency SOS button"}
            accessibilityHint={countdown ? "Tap to start emergency countdown, tap again to cancel" : "Tap to trigger immediate emergency alert"}
            testID={testID}
          >
            <Text style={{
              color: theme.colors.text.inverse,
              fontSize: countdownActive ? 24 : 18,
              fontWeight: 'bold',
              textAlign: 'center',
            }}>
              {countdownActive ? timeLeft : 'SOS'}
            </Text>
            {!countdownActive && (
              <Text style={{
                color: theme.colors.text.inverse,
                fontSize: 12,
                marginTop: 4,
              }}>
                Emergency
              </Text>
            )}
          </TouchableOpacity>
        </Animated.View>
      );
    };
    ```

    ```typescript
    // components/EmergencyAlert.tsx
    import React from 'react';
    import { View, Text, TouchableOpacity } from 'react-native';
    import { useTheme } from '../contexts/ThemeContext';
    import { EmergencyAlert as EmergencyAlertType } from '../types/emergency';
    import { formatDistanceToNow } from 'date-fns';

    interface EmergencyAlertProps {
      alert: EmergencyAlertType;
      onResolve?: (alertId: string) => void;
      onViewDetails?: (alertId: string) => void;
      compact?: boolean;
    }

    export const EmergencyAlert: React.FC<EmergencyAlertProps> = ({
      alert,
      onResolve,
      onViewDetails,
      compact = false,
    }) => {
      const { theme } = useTheme();
      
      const getAlertTypeLabel = (type: string) => {
        const labels = {
          manual: 'Manual Emergency',
          automated: 'Automatic Detection',
          duress: 'Duress Signal',
          check_in_missed: 'Missed Check-in',
        };
        return labels[type] || 'Emergency Alert';
      };
      
      const getStatusColor = (status: string) => {
        switch (status) {
          case 'active': return theme.colors.emergency;
          case 'acknowledged': return theme.colors.warning;
          case 'resolved': return theme.colors.success;
          default: return theme.colors.text.secondary;
        }
      };
      
      if (compact) {
        return (
          <View style={{
            flexDirection: 'row',
            alignItems: 'center',
            padding: theme.spacing.sm,
            backgroundColor: theme.colors.surface,
            borderLeftWidth: 4,
            borderLeftColor: getStatusColor(alert.status),
            marginVertical: theme.spacing.xs,
          }}>
            <View style={{ flex: 1 }}>
              <Text style={{
                fontSize: theme.typography.body.fontSize,
                fontWeight: 'bold',
                color: theme.colors.text.primary,
              }}>
                {getAlertTypeLabel(alert.alert_type)}
              </Text>
              <Text style={{
                fontSize: theme.typography.caption.fontSize,
                color: theme.colors.text.secondary,
              }}>
                {formatDistanceToNow(new Date(alert.created_at), { addSuffix: true })}
              </Text>
            </View>
            <View style={{
              backgroundColor: getStatusColor(alert.status),
              paddingHorizontal: theme.spacing.sm,
              paddingVertical: theme.spacing.xs,
              borderRadius: theme.borderRadius.sm,
            }}>
              <Text style={{
                color: theme.colors.text.inverse,
                fontSize: theme.typography.caption.fontSize,
                fontWeight: 'bold',
              }}>
                {alert.status.toUpperCase()}
              </Text>
            </View>
          </View>
        );
      }
      
      return (
        <View style={{
          backgroundColor: theme.colors.surface,
          borderRadius: theme.borderRadius.md,
          padding: theme.spacing.md,
          marginVertical: theme.spacing.sm,
          borderWidth: 2,
          borderColor: getStatusColor(alert.status),
          ...theme.shadows.md,
        }}>
          <View style={{
            flexDirection: 'row',
            justifyContent: 'space-between',
            alignItems: 'flex-start',
            marginBottom: theme.spacing.sm,
          }}>
            <View style={{ flex: 1 }}>
              <Text style={{
                fontSize: theme.typography.h2.fontSize,
                fontWeight: theme.typography.h2.fontWeight,
                color: theme.colors.text.primary,
                marginBottom: theme.spacing.xs,
              }}>
                {getAlertTypeLabel(alert.alert_type)}
              </Text>
              <Text style={{
                fontSize: theme.typography.caption.fontSize,
                color: theme.colors.text.secondary,
              }}>
                Created {formatDistanceToNow(new Date(alert.created_at), { addSuffix: true })}
              </Text>
            </View>
            <View style={{
              backgroundColor: getStatusColor(alert.status),
              paddingHorizontal: theme.spacing.md,
              paddingVertical: theme.spacing.sm,
              borderRadius: theme.borderRadius.md,
            }}>
              <Text style={{
                color: theme.colors.text.inverse,
                fontSize: theme.typography.body.fontSize,
                fontWeight: 'bold',
              }}>
                {alert.status.toUpperCase()}
              </Text>
            </View>
          </View>
          
          {alert.message && (
            <Text style={{
              fontSize: theme.typography.body.fontSize,
              color: theme.colors.text.primary,
              marginBottom: theme.spacing.md,
            }}>
              {alert.message}
            </Text>
          )}
          
          <View style={{
            flexDirection: 'row',
            justifyContent: 'space-between',
            alignItems: 'center',
          }}>
            {onViewDetails && (
              <TouchableOpacity
                onPress={() => onViewDetails(alert.id)}
                style={{
                  paddingHorizontal: theme.spacing.md,
                  paddingVertical: theme.spacing.sm,
                  borderWidth: 1,
                  borderColor: theme.colors.primary,
                  borderRadius: theme.borderRadius.sm,
                }}
              >
                <Text style={{
                  color: theme.colors.primary,
                  fontSize: theme.typography.body.fontSize,
                  fontWeight: 'bold',
                }}>
                  View Details
                </Text>
              </TouchableOpacity>
            )}
            
            {alert.status === 'active' && onResolve && (
              <TouchableOpacity
                onPress={() => onResolve(alert.id)}
                style={{
                  backgroundColor: theme.colors.success,
                  paddingHorizontal: theme.spacing.md,

                  fontWeight: 'bold',
                }}>
                  Mark Safe
                </Text>
              </TouchableOpacity>

        </View>
      );
    ```

```typescript Location Sharing Component
// components/LocationSharingCard.tsx
import React, { useState } from 'react';
import { View, Text, TouchableOpacity, Switch } from 'react-native';
import { useTheme } from '../contexts/ThemeContext';
import { useLocationSharing } from '../hooks/useLocationSharing';
import { Contact } from '../types/contact';

interface LocationSharingCardProps {

  isSharing: boolean;
  onToggleSharing: (contactId: string, enabled: boolean) => void;
  onConfigurePermissions: (contactId: string) => void;

export const LocationSharingCard: React.FC<LocationSharingCardProps> = ({

  isSharing,
  onToggleSharing,
  onConfigurePermissions,

    }

    <View style={{
      backgroundColor: theme.colors.surface,
      borderRadius: theme.borderRadius.md,
      padding: theme.spacing.md,
      marginVertical: theme.spacing.sm,

        width: 50,
        height: 50,

        backgroundColor: theme.colors.primary,
        justifyContent: 'center',
        alignItems: 'center',
        marginRight: theme.spacing.md,
      }}>
        <Text style={{
          color: theme.colors.text.inverse,
          fontSize: 18,
          fontWeight: 'bold',
        }}>
          {contact.name.charAt(0).toUpperCase()}
        </Text>
      </View>
      
      <View style={{ flex: 1 }}>
        <Text style={{
          fontSize: theme.typography.body.fontSize,
          fontWeight: 'bold',
          color: theme.colors.text.primary,

        }}>
          {contact.name}
        </Text>
        <Text style={{
          fontSize: theme.typography.caption.fontSize,
          color: theme.colors.text.secondary,
        }}>
          {contact.relationship || 'Emergency Contact'}
        </Text>
        
        {isSharing && (
          <TouchableOpacity
            onPress={() => onConfigurePermissions(contact.id)}
            style={{ marginTop: theme.spacing.xs }}
          >
            <Text style={{
              fontSize: theme.typography.caption.fontSize,
              color: theme.colors.primary,
              textDecorationLine: 'underline',
            }}>
              Configure permissions
            </Text>
          </TouchableOpacity>
        )}
      </View>
      
      <Switch
        value={isSharing}
        onValueChange={handleToggle}
        disabled={isLoading}
        trackColor={{ 
          false: theme.colors.text.secondary, 
          true: theme.colors.success 
        }}
        thumbColor={isSharing ? theme.colors.text.inverse : theme.colors.background}
      />
    </View>
  );
};
```

```typescript Real-time Location Display
// components/LocationDisplay.tsx
import React, { useEffect, useState } from 'react';
import { View, Text } from 'react-native';
import { useTheme } from '../contexts/ThemeContext';
import { useRealTimeLocation } from '../hooks/useRealTimeLocation';
import { formatDistanceToNow } from 'date-fns';

interface LocationDisplayProps {
  userId: string;
  showAccuracy?: boolean;
  showTimestamp?: boolean;
  compact?: boolean;
}

export const LocationDisplay: React.FC<LocationDisplayProps> = ({
  userId,
  showAccuracy = true,
  showTimestamp = true,
  compact = false,
}) => {
  const { theme } = useTheme();
  const { location, isLoading, error } = useRealTimeLocation(userId);
  
  if (isLoading) {
    return (
      <View style={{
        padding: theme.spacing.md,
        backgroundColor: theme.colors.surface,
        borderRadius: theme.borderRadius.md,
        alignItems: 'center',
      }}>
        <Text style={{
          color: theme.colors.text.secondary,
          fontSize: theme.typography.body.fontSize,
        }}>
          Loading location...
        </Text>
      </View>
    );
  }
  
  if (error || !location) {
    return (
      <View style={{
        padding: theme.spacing.md,
        backgroundColor: theme.colors.surface,
        borderRadius: theme.borderRadius.md,
        alignItems: 'center',
        borderWidth: 1,
        borderColor: theme.colors.warning,
      }}>
        <Text style={{
          color: theme.colors.warning,
          fontSize: theme.typography.body.fontSize,
          textAlign: 'center',
        }}>
          Location unavailable
        </Text>
      </View>
    );

    if (accuracy <= 5) return 'Very precise';
    if (accuracy <= 15) return 'Precise';
    if (accuracy <= 50) return 'Good';
    return 'Approximate';
  };
  
  const getAccuracyColor = (accuracy: number) => {
    if (accuracy <= 15) return theme.colors.success;
    if (accuracy <= 50) return theme.colors.warning;
    return theme.colors.error;
  };
  
  if (compact) {
    return (
      <View style={{
        flexDirection: 'row',
        alignItems: 'center',
        padding: theme.spacing.sm,
        backgroundColor: theme.colors.surface,
        borderRadius: theme.borderRadius.sm,
      }}>

          width: 8,
          height: 8,
          borderRadius: 4,
          backgroundColor: getAccuracyColor(location.accuracy || 999),
          marginRight: theme.spacing.sm,
        }} />
        <Text style={{
          fontSize: theme.typography.caption.fontSize,
          color: theme.colors.text.primary,
        }}>
          {location.latitude.toFixed(6)}, {location.longitude.toFixed(6)}
        </Text>
      </View>
    );
  }
  
  return (
    <View style={{
      backgroundColor: theme.colors.surface,
      borderRadius: theme.borderRadius.md,
      padding: theme.spacing.md,
      ...theme.shadows.sm,
    }}>
      <View style={{
        flexDirection: 'row',
        justifyContent: 'space-between',
        alignItems: 'center',
        marginBottom: theme.spacing.sm,
      }}>
        <Text style={{
          fontSize: theme.typography.h2.fontSize,
          fontWeight: 'bold',
          color: theme.colors.text.primary,
        }}>
          Current Location
        </Text>
        <View style={{
          flexDirection: 'row',
          alignItems: 'center',
        }}>
          <View style={{
            width: 8,
            height: 8,
            borderRadius: 4,
```
